# Datetime

Datetime is a utility component that provides a convenient wrapper around the [Intl.DateTimeFormat](https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/Intl/DateTimeFormat) API.

export const Default = (args) => {
    return html`
<div style="padding: 10%; display: flex; justify-content: center;">
    <rux-datetime
        date=${args.date}
        day=${args.day}
        hour=${args.hour}
        minute=${args.minute}
        month=${args.month}
        second=${args.second}
        time-zone=${args.timeZone}
        year=${args.year}
        hour-12=${args.hour12}
        .era=${args.era}
    ></rux-datetime>
</div>
    `
}
